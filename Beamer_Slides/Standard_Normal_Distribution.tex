\section{Find the z scores corresponding to a given area}

\begin{frame}{Finding z Score for a Given Area}
This is just working backwards from the our previous examples.	\newline\\	\pause

Keep in mind that most (not all) technology will ask you for the area to the \emph{left} of the needed z score.
\end{frame}

\begin{frame}{Example 4}
Find the z score which has the given area to the left of it. Round to 3 decimal places.	\newline\\
(a) \quad 0.05	\newline\\
\begin{minipage}{0.5\textwidth}
\onslide<2->{
\begin{tikzpicture}[scale=0.65]
\begin{axis}[no marks, samples=200, axis lines = left, xmin = -3.5, xmax = 3.5, ymax = 0.45, xticklabels = {},
xtick style={draw=none}, xlabel = {z score}, ylabel = {Probability}
]
\addplot [draw = none, fill = green, domain = -3.5:-1.645] {gauss(0,1)} \closedcycle;
\addplot [very thick, color=blue] {gauss(0,1)};
\draw [->, >=stealth, very thick] (axis cs:-2.5,0.2) -- (-2.05,0.015);
\node at (axis cs:-2.5,0.2) [above] {0.05};
\end{axis}
\end{tikzpicture}
}
\end{minipage}
\hspace{0.25cm}
\onslide<3->{
	\begin{minipage}{0.3\textwidth}
	$z \approx -1.645$
	\end{minipage}
}
\end{frame}
